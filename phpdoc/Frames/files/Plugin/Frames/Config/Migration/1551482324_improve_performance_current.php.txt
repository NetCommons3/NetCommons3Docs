class ImprovePerformanceCurrent extends NetCommonsMigration {

 * Migration description

/**

 */

			'drop_field' => array(

		),
		'down' => array(
			'create_field' => array(
				'frames' => array(
					'indexes' => array(

					),
				),
			),
		),
	);

	}

/**
 * After migration callback

	}
}
